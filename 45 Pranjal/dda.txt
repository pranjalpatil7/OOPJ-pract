#include <graphics.h>
#include <iostream>
#include <math.h>
#include <conio.h>

int main()
{
    float x, y, x1, y1, x2, y2, dx, dy, step;
    int i, gd = DETECT, gm;

    initgraph(&gd, &gm, "");

    std::cout << "Enter the value of x1 and y1: ";
    std::cin >> x1 >> y1;
    std::cout << "Enter the value of x2 and y2: ";
    std::cin >> x2 >> y2;

    dx = (x2 - x1);
    dy = (y2 - y1);

    if (fabs(dx) >= fabs(dy))
        step = fabs(dx);
    else
        step = fabs(dy);

    dx = dx / step;
    dy = dy / step;

    x = x1;
    y = y1;
    i = 0;

    while (i <= step) {
        putpixel(x, y, 5);
        x = x + dx;
        y = y + dy;
        i = i + 1;
        delay(100);
    }

    getch();
    closegraph();
    return 0;
}
